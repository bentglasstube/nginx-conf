server {
  listen 80;
  listen [::]:80;

  server_name rss.eatabrick.org;

  error_log /var/log/nginx/rss.eatabrick.org.error.log;

  root /srv/http/rss.eatabrick.org/public;
  location / {
    try_files $uri @proxy;
    expires max;
  }

  error_page 404              /404.html;
  error_page 500 502 503 504  /5xx.html;

  location @proxy {
    proxy_set_header Host $http_host;
    proxy_set_header X-Forwarded-Host $host;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_pass http://unix:/var/run/dancer/eab-rss.sock:;
  }
}
